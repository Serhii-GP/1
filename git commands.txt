# windows 
# ls - список файлів в цій папці
# dir - що є в цій папці
# cd - зміна папки на ...
# md mkdir - створення папки
# cd .. - повернутись назад на 1 рівень, cd ../../../ на три рівні
# rmdir - видалення папки
# rm - видалити файл
# rm -rf - видаляє усе
# touch - створення файлу
# nano - виклик редактора коду у середині консолі


# git

# git init - ініціалізація гіта в данній папці
# git add . - додати файли в гіт
# git status - перевірити статус файлів
# git commit -m 'first commin' - створення коментарю до змін
# git log - передивитись усі коміти та зміни


# git checkout 'номер комміту' - перейти на цей коміт
# git checkout - поверне на основну гілку
# git reset 'номер комміту' - повернення до цього комміту без видалення інших
# git reset --hard 'номер комміту' - повернення до цього комміту і видалення всього, що було після нього.
